async function loadDocs() { const container = document.getElementById("docs"); container.innerHTML = ""; try { const docs = await fetch("data/catalog.json").then(r => r.json()); docs.forEach(doc => { container.innerHTML += `

${doc.titre}

${doc.matiere} - ${doc.niveau} - ${doc.type}

Télécharger
`; }); } catch (e) { container.innerHTML = "

Aucun document trouvé

"; } }
